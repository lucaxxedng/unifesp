CREATE TABLE PAIS(
	ID INT,
	NOME VARCHAR(50),
	PRYMARY KEY (ID),
)

CREATE TABLE CIDADE(
	ID INT,
	NOME VARCHAR(50),
	IDPAIS INT,
	PRIMARY KEY (ID),
	FOREIGN KEY (IDPAIS) REFERENCES PAIS(ID),
)	


CREATE TABLE COMPANIAAREA(
	ID INT,
	NOME VARCHAR(50),
	PRIMARY KEY (ID),
)

CREATE TABLE VOO(
	ID INT,
	HORACHEGADA TIME,
	HORASAIDA   TIME,
	IDCIDADECHEGADA	INT,
	IDCIDADESAIDA	INT,
	IDCOMPANIAAREA	INT,
	PRIMARY KEY (ID),
	FOREIGN KEY(IDCIDADECHEGADA, IDCIDADESAIDA) REFERENCES CIDADE(ID),
	FOREIGN KEY(IDCOMPANIAAREA) REFERENCES COMPANIAAREA(ID),
)


CREATE TABLE PASSAGEIRO(
	ID INT,
	NOME VARCHAR(50),
	CPF INT,
	EMAIL VARCHAR(50),
	DATANASCIMENTO DATETIME,
	PASSAPORTE INT,
	IDPAIS	INT,
	PRIMARY KEY(ID),
	FOREIGN KEY(IDPAIS) REFERENCES PAIS(ID)
)

CREATE TABLE PASSAGEM(
	ID INT,
	ASSENTO VARCHAR(50),
	IDPASSAGEIRO INT,
	IDVOO	INT,

	PRIMARY KEY(ID),
	FOREIGN KEY (IDPASSAGEIRO) REFERENCES PASSAGEIRO(ID),
	FOREIGN KEY(IDVOO) REFERENCES VOO(ID),
)

CREATE TABLE MODELO(
	ID  INT,
	NOME VARCHAR(50),
	CAPACIDADE INT,
	
	PRIMARY KEY(ID),
)

CREATE TABLE AVIAO(
	ID  INT,
	HORASDEVOO INT,
	DATAFABRICACAO DATETIME,
	NUMESTACIONAMENTO INT,
	IDMODELO VARCHAR(50),
	PRIMARY KEY (ID),
	FOREIGN KEY(IDMODELO) REFERENCES MODELO(ID),
)

CREATE TABLE PILOTO(
	ID INT,
	NOME VARCHAR(50),
	CPF	INT,
	IDCOMPANIAAREA INT,
	PRIMARY KEY (ID),
	FOREIGN KEY(IDCOMPANIAAREA) COMPANIAAREA(ID),
)

CREATE TABLE AVIAOPILOTOVOO (
	IDAVIAO INT,
	IDVOO	INT,
	IDPILOTO INT,
	PRIMARY KEY(IDAVIAO, IDVOO, IDPILOTO),
	FOREIGN KEY(IDAVIAO) REFERENCES AVIAO(ID),
	FOREIGN KEY(IDVOO) REFERENCES VOO(ID),
	FOREIGN KEY(IDPILOTO) REFERENCES PILOTO(ID),
)

CREATE TABLE BAGAGEM (
	ID	INT,
	PESO FLOAT,
	IDPASSAGEIRO INT,
	PRIMARY KEY(ID),
	FOREIGN KEY(IDPASSAGEIRO) REFERENCES PASSAGEIRO(ID),
)

CREATE TABLE AEROPORTO (
	ID INT,
	NOME VARCHAR(50),
	IDCIDADE INT,
	PRIMARY KEY (ID),
	FOREIGN KEY (IDCIDADE) REFERENCES CIDADE(ID),
)

CREATE TABLE EQUIPE (
	ID 	INT,
	TIPO VARCHAR(50),
	PRIMARY KEY (ID),
)

CREATE TABLE FUNCIONARIO (
	ID INT,
	SALARIO FLOAT,
	NOME VARCHAR(50),
	JORNADA	DATETIME,
	NUMDEPENDENTES INT,
	IDEQUIPE INT,
	PRIMARY KEY (ID),
	FOREIGN KEY (IDEQUIPE) REFERENCES EQUIPE(ID),
)

CREATE TABLE MOTORISTA (
	IDFUNCIONARIO INT,
	CNH INT,
	PRYMARY KEY (IDFUNCIONARIO),
	FOREIGN KEY (IDFUNCIONARIO) REFERENCES FUNCIONARIO(ID),
)

CREATE TABLE TAXI (
	ID 	INT,
	PLACA VARCHAR(50),
	IDMOTORISTA INT,
	IDAEROPORTO INT,
	PRIMARY KEY (ID),
	FOREIGN KEY (IDMOTORISTA) REFERENCES MOTORISTA(ID),
	FOREIGN KEY (IDAEROPORTO) REFERENCES AEROPORTO(ID),
)

CREATE TABLE RAMO(
	ID INT,
	NOME VARCHAR(50),
	
	PRIMARY KEY(ID),
)

CREATE TABLE LOJA(
	ID INT,
	QNTFUNCIONARIOS INT,
	NOME VARCHAR(50),
	IDRAMO INT,
	IDAEROPORTO INT,
	PRIMARY(ID),
	FOREIGN KEY(IDRAMO) REFERENCES RAMO(ID),
	FOREIGN KEY(IDAEROPORTO) REFERENCES AEROPORTO(ID),
)
